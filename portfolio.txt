      // --- 1. PRELOADER ---

      // --- 2. SMOOTH SCROLL ---
      const lenis = new Lenis({

        orientation: "vertical",
        smoothWheel: true,

      function raf(time) {

      // --- 3. THREE.JS: QUANTUM GLOBE ---
      function initThree() {
        const canvas = document.querySelector("#webgl-canvas");
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
        
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

        // Group to hold everything
        const group = new THREE.Group();
        scene.add(group);

        // A. The Wireframe Globe
        const geometry = new THREE.IcosahedronGeometry(10, 2);
        const wireframeMaterial = new THREE.MeshBasicMaterial({ color: 0x333333, wireframe: true, transparent: true, opacity: 0.1 });
        const globe = new THREE.Mesh(geometry, wireframeMaterial);
        group.add(globe);

        // B. The Particle Cloud (Plexus Effect)
        const pGeo = new THREE.BufferGeometry();
        const pCount = 600;
        const posArray = new Float32Array(pCount * 3);
        
        for(let i=0; i<pCount*3; i++) {
          posArray[i] = (Math.random() - 0.5) * 25; // Scatter around center
        }
        pGeo.setAttribute('position', new THREE.BufferAttribute(posArray, 3));
        const pMat = new THREE.PointsMaterial({
          size: 0.05,
          color: 0xffffff,
          transparent: true,
          opacity: 0.6
        });
        const particles = new THREE.Points(pGeo, pMat);
        group.add(particles);

        // C. Connecting Lines (Visual Complexity)
        // Note: Real-time line calculation is heavy, so we use a static line sphere for performance
        const lineGeo = new THREE.WireframeGeometry(new THREE.IcosahedronGeometry(8, 1));
        const lineMat = new THREE.LineBasicMaterial({ color: 0xffffff, transparent: true, opacity: 0.05 });
        const innerLines = new THREE.LineSegments(lineGeo, lineMat);
        group.add(innerLines);

        camera.position.z = 20;

        // Mouse Interaction
        let mouseX = 0, mouseY = 0;
        let targetX = 0, targetY = 0;
        
        document.addEventListener('mousemove', (e) => {
          mouseX = (e.clientX - window.innerWidth / 2) * 0.001;
          mouseY = (e.clientY - window.innerHeight / 2) * 0.001;
        });

        const clock = new THREE.Clock();

        function animate() {
          const t = clock.getElapsedTime();

          targetX = mouseX * 2;
          targetY = mouseY * 2;

          // Smooth rotation
          group.rotation.y += 0.002;
          group.rotation.x += (targetY - group.rotation.x) * 0.05;
          group.rotation.y += (targetX - group.rotation.y) * 0.05;

          // Breathing effect
          const scale = 1 + Math.sin(t * 0.5) * 0.02;
          group.scale.set(scale, scale, scale);
          
          renderer.render(scene, camera);
          requestAnimationFrame(animate);
        }
        animate();

        window.addEventListener('resize', () => {
          camera.aspect = window.innerWidth / window.innerHeight;
          camera.updateProjectionMatrix();
          renderer.setSize(window.innerWidth, window.innerHeight);
        });
      }

          const originalText = event.target.dataset.scramble;
          let interval = setInterval(() => {
            event.target.innerText = originalText
              .split("")
              .map((letter, index) => {
                if(index < iteration) return originalText[index];
                return letters[Math.floor(Math.random() * 36)];
              })
              .join("");
            
            if(iteration >= originalText.length) clearInterval(interval);
            iteration += 1/3; 
          }, 30);
        });
      }

      function initScroll() {

          const wrapper = document.querySelector("#ventures-pin-wrapper");

          const scrollAmount = container.scrollWidth - window.innerWidth + window.innerWidth * 0.15;

              pin: true,
              scrub: 1,
              end: () => "+=" + scrollAmount,
              invalidateOnRefresh: true
            }
          });

      // --- 6. SMART NAVIGATION LOGIC ---
      function initSmartNav() {
        let lastScrollY = window.scrollY;
        const nav = document.getElementById('navbar');
        
        window.addEventListener('scroll', () => {
          const currentScrollY = window.scrollY;
          
          // If scrolling down AND passed a small threshold (50px)
          if (currentScrollY > lastScrollY && currentScrollY > 50) {
            nav.style.transform = 'translateY(-100%)';
          } else {
            // If scrolling up OR at the top
            nav.style.transform = 'translateY(0)';
          }
          
          lastScrollY = currentScrollY;
        });
      }

        el.addEventListener("mouseleave", () => cursorCircle.classList.remove("active"));
      });
    </script>
